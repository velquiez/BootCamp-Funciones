import javax.swing.*;
import javax.swing.event.*;
import java.awt.*;

public class Redondeos extends JFrame implements ActionListener
{
 private JTextField textfield;
 private JLabel titulo, ingrese, floor, ceil, round;
 private JButton Redondear;
 double num = 0;
 double rr = 0;
 double rc = 0;
 double rf = 0; 

 public Redondeos()
 {
  setLayout(null);
   setTitle("Metodos para redondeos");
   getContentPane().setBackground(new Color(89,0,0));

  titulo = new JLabel("Metodos para aplicar redondeos");
   titulo.setBounds(5,10,200,30);
   add(titulo);

  ingrese = new JLabel("Ingrese n√∫mero a redondear: ");
   ingrese.setBounds(15,50,200,30);
   add(ingrese);

  textfield = new JTextField();
   textfield.setBounds(15,75,150,30);
   add(textfield);

  floor = new JLabel("rf = Math.floor(num): ");
   floor.setBounds(5,90,200,30);
   add(floor);

  ceil = new JLabel("rc = Math.ceil(num): ");
   ceil.setBounds(5,120,200,30);
   add(ceil);

  round = new JLabel("rr = Math.round(num): ");
   round.setBounds(5,150,200,30);
   add(round);  
 } 
  public void actionPerformed(ActionEvent e)
 {
  if(e.getSource() == ingresar)
    {
     num = textfield.getText();

     rr = Math.round(num);
     rc = Math.ceil(num);
     rf = Math.floor(num);

     floor.setText("rf = Math.floor(num): " + rf);
     ceil.setText("rc = Math.ceil(num): " + rc);
     round.setText("rr = Math.round(num): " + rr);
    }
 }
